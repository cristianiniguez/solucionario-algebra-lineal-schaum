a. $x(3, 2) = 2(y, -1)$

$$
\begin{align*}
 x(3, 2) &= 2(y, -1) \\
(3x, 2x) &= (2y, -2) \\
\end{align*}
$$

$$
\begin{align*}
2x &= -2 \\
 x &= -1 \\
\end{align*}
$$

$$
\begin{align*}
   3x &= 2y \\
3(-1) &= 2y \\
    y &= -\frac{3}{2} \\
\end{align*}
$$

b. $x(2, y) = y(1, -2)$

$$
\begin{align*}
 x(2, y) &= y(1, -2) \\
(2x, xy) &= (y, -2y) \\
\end{align*}
$$

$$
\begin{align*}
2x &= y \\
 y &= 2x \\
\end{align*}
$$

$$
\begin{align*}
      xy &= -2y \\
   x(2x) &= -2(2x) \\
    2x^2 &= -4x \\
x^2 + 2x &= 0 \\
x(x + 2) &= 0 \\
     x_1 &= 0 \\
     x_2 &= -2 \\
\end{align*}
$$

$$
\begin{align*}
y_1 &= 2x_1 = 2(0) = 0 \\
y_2 &= 2x_2 = 2(-2) = -4 \\
